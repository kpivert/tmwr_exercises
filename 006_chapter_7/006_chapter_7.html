1.10 Using multilevelmod

require(multilevelmod)
Loading required package: multilevelmod
multilevel_spec <- 
  linear_reg() |> 
  set_engine("lmer")

multilevel_workflow <- 
  workflow() |> 
  # Pass the data along as-is:
  add_variables(
    outcomes = distance, 
    predictors = c(Sex, age, Subject)
    ) |> 
  add_model(
    multilevel_spec,
    # this formulat is passed to the model
    formula = distance ~ Sex + (age | Subject)

multilevel_fit <-
